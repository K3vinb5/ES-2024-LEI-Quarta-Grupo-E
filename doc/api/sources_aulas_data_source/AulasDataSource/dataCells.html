List<DataCell> dataCells(ClassModel aula) {
  List<DataCell> returnList = [];
  for(int i = 0; i < aula.getPropertiesList().length; i++){
    if(visibleColumns[i]){
      returnList.add(DataCell(Text(aula.getPropertiesList()[i])));
    }
  }
  returnList.add(DataCell(Text(aula.calculateYearWeek(aula.date))));
  returnList.add(DataCell(Text(aula.calculateSemesterWeek(aula.date))));
  return returnList;
}
